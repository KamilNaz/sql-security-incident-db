Sample Queries

Basic Incident Retrieval

Get All Critical Incidents 
SELECT
    IncidentID,
    Timestamp,
    IncidentType,
    SourceIP,
    DestinationIP,
    Status,
    AssignedAnalyst
FROM Incidents
WHERE Severity = 'CRITICAL'
ORDER BY Timestamp DESC;

Incident Statistics by Severity

Count Incidents by Severity 
SELECT
    Severity,
    COUNT(*) AS IncidentCount,
    ROUND(COUNT(*) * 100.0 / SUM(COUNT(*)) OVER(), 2) AS Percentage
FROM Incidents
GROUP BY Severity
ORDER BY
    CASE Severity
        WHEN 'CRITICAL' THEN 1
        WHEN 'HIGH' THEN 2
        WHEN 'MEDIUM' THEN 3
        WHEN 'LOW' THEN 4
    END;

Analyst Workload Analysis

Analyst Performance Metrics 
SELECT
    A.AnalystName,
    A.Department,
    COUNT(I.IncidentID) AS TotalIncidents,
    SUM(CASE WHEN I.Status = 'Closed' THEN 1 ELSE 0 END) AS ClosedIncidents,
    SUM(CASE WHEN I.Status IN ('In Progress', 'Investigating') THEN 1 ELSE 0 END) AS ActiveIncidents,
    ROUND(
        SUM(CASE WHEN I.Status = 'Closed' THEN 1.0 ELSE 0 END) / COUNT(I.IncidentID) * 100,
        2
    ) AS ClosureRate
FROM Analysts A
LEFT JOIN Incidents I ON A.AnalystID = I.AssignedAnalyst
GROUP BY A.AnalystName, A.Department
ORDER BY TotalIncidents DESC;

Threat Intelligence Correlation

Match Incidents with Known Threats 
SELECT
    I.IncidentID,
    I.Timestamp,
    I.IncidentType,
    I.SourceIP,
    TI.ThreatLevel,
    TI.Source AS ThreatSource,
    TI.FirstSeen AS ThreatFirstSeen
FROM Incidents I
INNER JOIN ThreatIntelligence TI
    ON I.SourceIP = TI.IOC
WHERE TI.Active = 1
ORDER BY
    CASE TI.ThreatLevel
        WHEN 'CRITICAL' THEN 1
        WHEN 'HIGH' THEN 2
        WHEN 'MEDIUM' THEN 3
        ELSE 4
    END,
    I.Timestamp DESC;

Analytics Examples

Daily Incident Trends

Incident Volume Over Time 
SELECT
    CAST(Timestamp AS DATE) AS IncidentDate,
    COUNT(*) AS TotalIncidents,
    SUM(CASE WHEN Severity IN ('CRITICAL', 'HIGH') THEN 1 ELSE 0 END) AS HighSeverityIncidents,
    SUM(CASE WHEN Status = 'Closed' THEN 1 ELSE 0 END) AS ClosedIncidents
FROM Incidents
WHERE Timestamp >= DATEADD(day, -30, GETDATE())
GROUP BY CAST(Timestamp AS DATE)
ORDER BY IncidentDate DESC;

Top Attack Types

Most Common Attack Patterns 
SELECT
    IncidentType,
    COUNT(*) AS Occurrences,
    AVG(CASE
        WHEN Severity = 'CRITICAL' THEN 4
        WHEN Severity = 'HIGH' THEN 3
        WHEN Severity = 'MEDIUM' THEN 2
        ELSE 1
    END) AS AvgSeverityScore
FROM Incidents
GROUP BY IncidentType
ORDER BY Occurrences DESC
LIMIT 10;

Incident Response Time

Mean Time to Resolution (MTTR) 
WITH IncidentTimeline AS (
    SELECT
        I.IncidentID,
        I.Timestamp AS OpenTime,
        MAX(IA.Timestamp) AS CloseTime,
        I.Severity
    FROM Incidents I
    LEFT JOIN IncidentActions IA ON I.IncidentID = IA.IncidentID
    WHERE I.Status = 'Closed'
    GROUP BY I.IncidentID, I.Timestamp, I.Severity
)
SELECT
    Severity,
    COUNT(*) AS ClosedIncidents,
    AVG(DATEDIFF(minute, OpenTime, CloseTime)) AS AvgResolutionMinutes,
    MIN(DATEDIFF(minute, OpenTime, CloseTime)) AS MinResolutionMinutes,
    MAX(DATEDIFF(minute, OpenTime, CloseTime)) AS MaxResolutionMinutes
FROM IncidentTimeline
GROUP BY Severity
ORDER BY
    CASE Severity
        WHEN 'CRITICAL' THEN 1
        WHEN 'HIGH' THEN 2
        WHEN 'MEDIUM' THEN 3
        WHEN 'LOW' THEN 4
    END;

Quick Start

Installation

Choose your database platform and follow the installation steps:

SQL Server 
sqlcmd -S localhost -d SecurityDB -i demo_setup.sql
MySQL 
mysql -u root -p SecurityDB < demo_setup.sql
PostgreSQL 
psql -d securitydb -f demo_setup.sql

Running Analytics

After loading the demo data, you can run the built-in analytics queries:

Execute Trend Analysis 
sqlcmd -S localhost -d SecurityDB -i analytics/trend_analysis.sql
